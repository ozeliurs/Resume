\resumeItemListStart
\vspace{0.5pt}
\resumeItem{
    \textbf{ByteEats}, a food delivery app with multi-restaurant ordering, payment integrations, real-time updates, and cancellation policies. It provides statistical insights for users and administrators, and allows restaurants to manage their operations efficiently. \\
    \vspace{-5pt}
    \begin{itemize}
        \item \textbf{Design:} Prior to development, an architecture report was prepared including personas, a use case diagram, and a class diagram.
        \item \textbf{Project Management:} Extensive use of GitHub issues for tracking development progress, with user stories and features translated into Gherkin tests.
        \item \textbf{CI/CD:} GitHub Actions for automating Maven builds, releases with semantic versioning, testing with JUnit and Cucumber, and code quality checks with SonarQube.
    \end{itemize}
}
\vspace{-5pt}

\resumeItem{
    \textbf{Webonjour}, a quizzer app designed with elerly people in mind, especially those with Alzheimer's disease. It provides a simple and intuitive interface, with a focus on accessibility and user experience. \\
    \vspace{-5pt}
    \begin{itemize}
        \item \textbf{Design:} Extensive use of Figma for prototyping, with reviews by an occupational therapist.
        Personas and user stories based on needs gathered from the occupational therapist and indirectly from patients.
        \item \textbf{Project Development:} Angular app built with nx monorepo, rxjs, prisma and an express backend.
        \item \textbf{Testing:} Use of playwright for end-to-end testing, and jest for unit testing integrated in the CI/CD pipeline.
        \item \textbf{CI/CD:} GitHub Actions for automating builds, testing, and deployment to a VPS with Docker Compose.
    \end{itemize}
}
\vspace{-5pt}

\resumeItem{
    \textbf{Restorater}, a google maps clone to find restaurants. Lets business owners get verified reviews and ratings and gain visibility. Allows users to discover new restaurants and share their experiences. \\
    \vspace{-5pt}
    \begin{itemize}
        \item \textbf{Design:} Market analysis and personas were used to design the app, with a focus on user experience and accessibility.
        Prototypes were created using Figma and reviewed by other students.
        \item \textbf{Project Architecture:} Android app written in Kotlin with an MVVM architecture, using Google Maps API and a custom Backend to handle user reviews, ratings, and restaurant information.
    \end{itemize}
}
\vspace{-5pt}

\resumeItem{
    \textbf{Takenoko}, a recreation of the board game \href{https://www.espritjeu.com/jeu-de-societe/takenoko.html}{Takenoko} in Java.
    The game is a competitive strategy game where players take the role of gardeners, growing bamboo and feeding a panda.
    The game features an AI to play against. \\
    \vspace{-5pt}
    \begin{itemize}
        \item \textbf{Testing:} JUnit was used for unit testing, PIT for mutation testing, and JaCoCo for code coverage.
        Everything was integrated into the CI/CD pipeline and pushed to a SonarQube server for code quality checks.
    \end{itemize}
}
